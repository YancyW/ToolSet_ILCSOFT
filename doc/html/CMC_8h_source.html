00434                                 // get the angle between MC and other particle, then rank the angle from small to large
00435                                 std::vector<std::pair<float,MCParticle*> >Ref_angle;
00436                                 for(unsigned int i=0;i<Ref.size();i++){
00437                                         if(MC==Ref[i]){continue;}

00439                                         Ref_angle.push_back(pair_tmp);
00440                                 }
00441                                 std::sort(Ref_angle.begin(),Ref_angle.end(),Compare_as_Pair);
00442                                 for(unsigned int i=0;i<Ref_angle.size();i++){
00443                                 }
00444 
00445                                 // make a cone around the input MC, if the sum of the energies inside the cone is larger than the minimal_energy_in_the_cone, give the angle value
00446                                 float sum_energy=0;
00447                                 float angle=0;
00448                                 for(unsigned int i=0;i<Ref_angle.size();i++){
00449                                         sum_energy = sum_energy + Ref_angle[i].second->getEnergy();
00450                                         angle=Ref_angle[i].first;
00451                                         if(sum_energy>minimal_energy_in_the_cone){
00452                                                 if(angle>minimal_angle_from_input){
00453                                                         return(angle-0.01);
00454                                                 }
00455                                                 else{

00458                                         }
