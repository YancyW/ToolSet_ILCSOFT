00420                                 // get the angle between MC and other particle, then rank the angle from small to large
00421                                 std::vector<std::pair<float,MCParticle*> >Ref_angle;
00422                                 for(unsigned int i=0;i<Ref.size();i++){
00423                                         if(MC==Ref[i]){continue;}
00424                                         std::pair<float,MCParticle*> pair_tmp = std::make_pair(Cal_Angle_Deg(MC,Ref[i]),Ref[i]);
00425                                         Ref_angle.push_back(pair_tmp);
00426                                 }
00427                                 std::sort(Ref_angle.begin(),Ref_angle.end(),Compare_as_Pair);
00428                                 for(unsigned int i=0;i<Ref_angle.size();i++){
00429                                 }
00430 
00431                                 // make a cone around the input MC, if the sum of the energies inside the cone is larger than the minimal_energy_in_the_cone, give the angle value
00432                                 float sum_energy=0;
00433                                 float angle=0;
00434                                 for(unsigned int i=0;i<Ref_angle.size();i++){
00435                                         sum_energy = sum_energy + Ref_angle[i].second->getEnergy();
00436                                         angle=Ref_angle[i].first;
00437                                         if(sum_energy>minimal_energy_in_the_cone){
00438                                                 if(angle>minimal_angle_from_input){
00439                                                         return(angle-0.01);
00440                                                 }
00441                                                 else{
00442                                                         return(angle);

00444                                         }
00445                                 }
